package middleware

import (
	"net/http"
	"github.com/Bekican/gorenel/internal/server"
)

// Advanced Rate Limiting Middleware
// Extracts UserID from context (set by AuthMiddleware)
// and checks if they have remaining quota.

func RateLimitMiddleware(rl *server.AdvancedRateLimiter) func(http.HandlerFunc) http.HandlerFunc {
	return func(next http.HandlerFunc) http.HandlerFunc {
		return func(w http.ResponseWriter, r *http.Request) {
			// 1. Get UserID from Context (Set during Auth)
			userClaims := GetUserFromContext(r.Context())
			
			var identifier string
			if userClaims != null {
				identifier = userClaims.UserID
			} else {
				// Fallback to IP if not logged in
				identifier = r.RemoteAddr 
			}

			// 2. Check Rate Limit
			if !rl.Allow(identifier) {
				w.Header().Set("Retry-After", "60")
				http.Error(w, "Too Many Requests: Quota Exceeded", http.StatusTooManyRequests)
				return
			}

			// 3. Continue
			next(w, r)
		}
	}
}
