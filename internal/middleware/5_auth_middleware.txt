package middleware

import (
	"context"
	"net/http"

	"github.com/Bekican/gorenel/pkg/auth"
	"github.com/Bekican/gorenel/pkg/errors"
)

type userContextKey string

const UserKey userContextKey = "user"

// RequireAuth middleware protects routes
func RequireAuth(jwtSvc *auth.JWTService) func(http.HandlerFunc) http.HandlerFunc {
	return func(next http.HandlerFunc) http.HandlerFunc {
		return func(w http.ResponseWriter, r *http.Request) error {
			// 1. Get token from cookie
			cookie, err := r.Cookie("auth_token")
			if err != nil {
				return errors.Unauthorized("Missing authentication token")
			}

			// 2. Validate token
			claims, err := jwtSvc.ValidateToken(cookie.Value)
			if err != nil {
				return errors.Unauthorized("Invalid or expired session")
			}

			// 3. Store user info in context
			ctx := context.WithValue(r.Context(), UserKey, claims)
			
			// 4. Update request with new context and call next handler
			return next(w, r.WithContext(ctx))
		}
	}
}

// GetUserFromContext helper to retrieve user from request context
func GetUserFromContext(ctx context.Context) *auth.Claims {
	user, ok := ctx.Value(UserKey).(*auth.Claims)
	if !ok {
		return nil
	}
	return user
}
