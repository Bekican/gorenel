package server

import (
	"io"
	"log"
	"net"

	"github.com/hashicorp/yamux"
)

// --- TCP Proxy Logic ---
func (p *TCPProxy) StartProxy(publicPort int, session *yamux.Session) {
	listener, err := net.Listen("tcp", fmt.Sprintf(":%d", publicPort))
	if err != nil {
		log.Printf("TCP Proxy başlatılamadı: %v", err)
		return
	}
	defer listener.Close()

	for {
		conn, err := listener.Accept()
		if err != nil {
			return
		}

		// Her bağlantı için yamux'tan yeni bir stream aç
		stream, err := session.OpenStream()
		if err != nil {
			conn.Close()
			continue
		}

		go func() {
			defer conn.Close()
			defer stream.Close()
			go io.Copy(stream, conn)
			io.Copy(conn, stream)
		}()
	}
}
