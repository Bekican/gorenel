package errors

import (
	"fmt"
	"net/http"
)

// AppError is our custom error type.
// It implements the standard error interface.
type AppError struct {
	Code    int    `json:"code"`    // HTTP Status Code (e.g., 400, 404, 500)
	Message string `json:"message"` // User-friendly message
	Err     error  `json:"-"`       // Internal error (not exposed to user)
}

// Implement the error interface
func (e *AppError) Error() string {
	if e.Err != nil {
		return fmt.Sprintf("%s: %v", e.Message, e.Err)
	}
	return e.Message
}

// Unwrap helps in error checking with errors.Is/As
func (e *AppError) Unwrap() error {
	return e.Err
}

// New creates a generic AppError
func New(code int, message string, err error) *AppError {
	return &AppError{
		Code:    code,
		Message: message,
		Err:     err,
	}
}

// Specific Error Helpers (Use these in your handlers)

func NewBadRequest(message string) *AppError {
	return New(http.StatusBadRequest, message, nil)
}

func NewNotFound(message string) *AppError {
	return New(http.StatusNotFound, message, nil)
}

func NewUnauthorized(message string) *AppError {
	return New(http.StatusUnauthorized, message, nil)
}

func NewInternalError(err error) *AppError {
	// Usually we hide the internal error message from user
	return New(http.StatusInternalServerError, "Internal server error", err)
}
