package errors

import (
	"fmt"
	"net/http"
	"runtime"
	"strings"
)

// ErrorType defines the category of the error
type ErrorType string

const (
	TypeValidation   ErrorType = "VALIDATION_ERROR"
	TypeNotFound     ErrorType = "NOT_FOUND"
	TypeUnauthorized ErrorType = "UNAUTHORIZED"
	TypeInternal     ErrorType = "INTERNAL_ERROR"
	TypeConflict     ErrorType = "CONFLICT"
)

// AppError is a production-grade error struct.
// It separates what the user sees (Message) from what we log (Err/StackTrace).
type AppError struct {
	Type       ErrorType         `json:"type"`              // Machine-readable error type
	Code       int               `json:"code"`              // HTTP Status Code
	Message    string            `json:"message"`           // Human-readable message for frontend
	FieldErrors map[string]string `json:"fields,omitempty"` // For validation (e.g., "email": "invalid format")
	Err        error             `json:"-"`                 // The original internal error (hidden from user)
	stack      string            // Stack trace for internal debugging
}

// Error implements the standard error interface
func (e *AppError) Error() string {
	return e.Message
}

// Unwrap allows checking the underlying error
func (e *AppError) Unwrap() error {
	return e.Err
}

// StackTrace returns the captured stack trace
func (e *AppError) StackTrace() string {
	return e.stack
}

// captureStack captures the call stack for debugging
func captureStack() string {
	var sb strings.Builder
	// Skip 3 frames: captureStack -> New -> Wrapper functions
	for i := 3; i < 15; i++ {
		_, file, line, ok := runtime.Caller(i)
		if !ok {
			break
		}
		// Clean up file path to be relative or shorter
		if idx := strings.LastIndex(file, "/gorenel/"); idx != -1 {
			file = file[idx+1:]
		}
		sb.WriteString(fmt.Sprintf("%s:%d\n", file, line))
	}
	return sb.String()
}

// --- Factory Functions ---

// New creates a new AppError with stack trace
func New(errType ErrorType, code int, msg string, cause error) *AppError {
	return &AppError{
		Type:    errType,
		Code:    code,
		Message: msg,
		Err:     cause,
		stack:   captureStack(),
	}
}

// NotFound creates a standard 404 error
func NotFound(message string, cause error) *AppError {
	return New(TypeNotFound, http.StatusNotFound, message, cause)
}

// BadRequest creates a 400 error.
func BadRequest(message string, cause error) *AppError {
	return New(TypeValidation, http.StatusBadRequest, message, cause)
}

// Internal wraps an unexpected error (500)
// The 'cause' will be logged but 'message' will be generic if not provided
func Internal(cause error) *AppError {
	return New(TypeInternal, http.StatusInternalServerError, "Bir hata oluştu, lütfen daha sonra tekrar deneyin.", cause)
}

// ValidationError creates a structured validation error (422)
// Example: fields = {"email": "invalid", "password": "too short"}
func ValidationError(message string, fields map[string]string) *AppError {
	err := New(TypeValidation, http.StatusUnprocessableEntity, message, nil)
	err.FieldErrors = fields
	return err
}

// Unauthorized creates a 401 error
func Unauthorized(message string) *AppError {
	return New(TypeUnauthorized, http.StatusUnauthorized, message, nil)
}
