package logger

import (
	"testing"

	"github.com/stretchr/testify/assert"
	"github.com/stretchr/testify/require"
	"go.uber.org/zap"
	"go.uber.org/zap/zaptest"
)

func TestInit(t *testing.T) {
	cfg := &Config{
		Level:       "debug",
		OutputPath:  "stdout",
		Development: true,
		Encoding:    "json",
	}

	err := Init(cfg)
	require.NoError(t, err, "Init should not return error")
	assert.NotNil(t, Log, "Log should be initialized")
}

func TestDefaultConfig(t *testing.T) {
	cfg := DefaultConfig()

	assert.Equal(t, "info", cfg.Level)
	assert.Equal(t, "stdout", cfg.OutputPath)
	assert.Equal(t, "json", cfg.Encoding)
	assert.False(t, cfg.Development)
}

func TestWithRequestID(t *testing.T) {
	// Use zaptest for testing
	Log = zaptest.NewLogger(t)

	childLogger := WithRequestID("test-123")
	assert.NotNil(t, childLogger)
}

func TestWithUserID(t *testing.T) {
	Log = zaptest.NewLogger(t)

	childLogger := WithUserID("user-456")
	assert.NotNil(t, childLogger)
}

func TestWithFields(t *testing.T) {
	Log = zaptest.NewLogger(t)

	childLogger := WithFields(
		zap.String("key1", "value1"),
		zap.Int("key2", 42),
	)
	assert.NotNil(t, childLogger)
}

func TestLogLevels(t *testing.T) {
	Log = zaptest.NewLogger(t)

	// These should not panic
	assert.NotPanics(t, func() {
		Debug("debug message", zap.String("key", "value"))
	})

	assert.NotPanics(t, func() {
		Info("info message", zap.Int("count", 10))
	})

	assert.NotPanics(t, func() {
		Warn("warn message")
	})

	assert.NotPanics(t, func() {
		Error("error message", zap.Error(nil))
	})
}

func TestSync(t *testing.T) {
	Log = zaptest.NewLogger(t)

	err := Sync()
	assert.NoError(t, err)
}

func TestSyncNilLogger(t *testing.T) {
	// Reset Log to nil
	originalLog := Log
	Log = nil
	defer func() { Log = originalLog }()

	err := Sync()
	assert.NoError(t, err)
}
