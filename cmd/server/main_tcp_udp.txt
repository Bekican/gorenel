package main

// --- cmd/server/main.go (handleClient içindeki REGISTER bloğu güncellemesi) ---

func handleClient(conn net.Conn, tm *server.TunnelManager, authManager *server.AuthManager) {
	// ... (RegisterRequest okuma ve Auth) ...

	if regReq.TunnelType == "tcp" || regReq.TunnelType == "udp" {
		// Port tahsis et
		publicPort, err := tm.AllocatePort()
		if err != nil {
			protocol.WriteMessage(conn, protocol.NewErrorMessage(500, "Port kalmadı"))
			return
		}

		// Dinleyiciyi başlat
		if regReq.TunnelType == "tcp" {
			go server.StartTCPProxy(publicPort, session)
		} else {
			go server.StartUDPProxy(publicPort, session)
		}

		// Yanıt gönder
		response := protocol.RegisterResponse{
			PublicPort: publicPort,
		}
		protocol.WriteMessage(conn, protocol.NewRegisterResponse(response))
		
		log.Printf("Raw %s tunnel created on port %d", regReq.TunnelType, publicPort)
	} else {
		// ... mevcut HTTP mantığı ...
	}
}
